// ==================== 4. PageResult（统一分页，不可修改） ====================
package com.microcourse.dto;

import com.baomidou.mybatisplus.core.metadata.IPage;
import lombok.Data;
import java.util.List;

@Data
public class PageResult<T> {
    private List<T> items;           // 当前页数据
    private int page;                // 当前页码（0-based）
    private int size;                // 每页条数
    private long totalElements;      // 总记录数
    private long totalPages;         // 总页数（自动计算）

    public static <T> PageResult<T> of(IPage<T> page) {
        PageResult<T> result = new PageResult<>();
        result.setItems(page.getRecords());
        result.setPage((int) page.getCurrent());
        result.setSize((int) page.getSize());
        result.setTotalElements(page.getTotal());
        result.setTotalPages(page.getPages());
        return result;
    }
}

// ==================== 5. R（统一响应，不可修改） ====================
package com.microcourse.dto;

import com.fasterxml.jackson.annotation.JsonInclude;
import com.microcourse.exception.ErrorCode;
import lombok.Data;

@Data
@JsonInclude(JsonInclude.Include.NON_NULL)
public class R<T> {
    private int code;               // 200 = 成功
    private String message;         // "ok" = 成功
    private T data;                 // 业务数据，失败时为 null
    private long timestamp;         // 毫秒时间戳

    public static <T> R<T> ok(T data) {
        R<T> r = new R<>();
        r.code = 200;
        r.message = "ok";
        r.data = data;
        r.timestamp = System.currentTimeMillis();
        return r;
    }

    public static <T> R<T> ok() {
        return ok(null);
    }

    public static <T> R<T> fail(int code, String message) {
        R<T> r = new R<>();
        r.code = code;
        r.message = message;
        r.timestamp = System.currentTimeMillis();
        return r;
    }

    public static <T> R<T> fail(ErrorCode errorCode) {
        return fail(errorCode.getCode(), errorCode.getMessage());
    }
}
